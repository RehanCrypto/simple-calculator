<body>
  <div class="calculator">
    <input type="text" id="input1" placeholder="Enter number 1" />
    <input type="text" id="input2" placeholder="Enter number 2" />
    <button onclick="add()">Add</button>
    <button onclick="subtract()">Subtract</button>
    <button onclick="multiply()">Multiply</button>
    <button onclick="divide()">Divide</button>
    <p id="result"></p>
  </div>

  <script>
    function add() {
      var num1 = parseFloat(document.getElementById("input1").value);
      var num2 = parseFloat(document.getElementById("input2").value);
      var result = num1 + num2;
      document.getElementById("result").innerHTML = "Result: " + result;
    }

    function subtract() {
      var num1 = parseFloat(document.getElementById("input1").value);
      var num2 = parseFloat(document.getElementById("input2").value);
      var result = num1 - num2;
      document.getElementById("result").innerHTML = "Result: " + result;
    }

    function multiply() {
      var num1 = parseFloat(document.getElementById("input1").value);
      var num2 = parseFloat(document.getElementById("input2").value);
      var result = num1 * num2;
      document.getElementById("result").innerHTML = "Result: " + result;
    }

    function divide() {
      var num1 = parseFloat(document.getElementById("input1").value);
      var num2 = parseFloat(document.getElementById("input2").value);
      var result = num1 / num2;
      document.getElementById("result").innerHTML = "Result: " + result;
    }
  </script>
